Real-World Problem – Nearest Parking Spot using Min Heap

This project demonstrates an efficient solution to the common problem of finding the nearest available parking spot in a parking lot. Using a min-heap data structure, we can quickly retrieve the closest parking spot based on distance from the entrance.

parking_spot.c
#include <stdio.h>
#include <stdlib.h>
#define MAX_SIZE 100  
typedef struct {
    int distance;
    int spotNumber;
} ParkingSpot;
typedef struct {
    ParkingSpot spots[MAX_SIZE];
    int size;
} MinHeap;
void swap(ParkingSpot *a, ParkingSpot *b) {
    ParkingSpot temp = *a;
    *a = *b;
    *b = temp;
}
void heapifyDown(MinHeap *heap, int index) {
    int smallest = index;
    int left = 2 * index + 1;
    int right = 2 * index + 2;
    if (left < heap->size && heap->spots[left].distance < heap->spots[smallest].distance)
        smallest = left;
    if (right < heap->size && heap->spots[right].distance < heap->spots[smallest].distance)
        smallest = right;
    if (smallest != index) {
        swap(&heap->spots[index], &heap->spots[smallest]);
        heapifyDown(heap, smallest);
    }
}
void insertSpot(MinHeap *heap, int distance, int spotNumber) {
    if (heap->size >= MAX_SIZE) {
        printf("Heap is full!\\n");
        return;
    }
    int i = heap->size;
    heap->spots[i].distance = distance;
    heap->spots[i].spotNumber = spotNumber;
    heap->size++;
    while (i != 0 && heap->spots[(i - 1) / 2].distance > heap->spots[i].distance) {
        swap(&heap->spots[i], &heap->spots[(i - 1) / 2]);
        i = (i - 1) / 2;
    }
}
ParkingSpot getNearestSpot(MinHeap *heap) {
    if (heap->size == 0) {
        printf("No available spots!\\n");
        return (ParkingSpot){-1, -1};
    }
    ParkingSpot nearest = heap->spots[0];
    heap->spots[0] = heap->spots[heap->size - 1];
    heap->size--;
    heapifyDown(heap, 0);
    return nearest;
}
int main() {
    MinHeap parkingLot = {.size = 0};
    insertSpot(&parkingLot, 5, 101);
    insertSpot(&parkingLot, 2, 102);
    insertSpot(&parkingLot, 8, 103);
    insertSpot(&parkingLot, 3, 104);
    printf("Nearest available spot: Spot %d at distance %d\\n", 
        getNearestSpot(&parkingLot).spotNumber, 
        getNearestSpot(&parkingLot).distance);
    return 0;
}
